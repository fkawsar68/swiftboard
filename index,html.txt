<script>

// ── STATE ──
let state = {
  boardTitle: 'My Board',
  columns: [],
  isPremium: false,

  nextColId: 1,
  nextCardId: 1
};

let dragCard = null, dragCol = null, editingCard = null, editingColId = null;

// ── PERSISTENCE ──
function save() {
  localStorage.setItem('swiftboard_v1', JSON.stringify(state));
}

function load() {
  try {
    const raw = localStorage.getItem('swiftboard_v1');
    if (raw) {
      state = JSON.parse(raw);
      return;
    }
  } catch(e) {}

  // Default board
  state.columns = [
    { id: 1, name: 'To Do', color: COL_COLORS[0], cards: [
      { id: 1, title: 'Welcome to SwiftBoard', desc: 'Drag cards between columns. Click + to add new cards.', priority: 'none', tag: 'start', due: '' },
      { id: 2, title: 'Add your first task', desc: '', priority: 'high', tag: '', due: '' }
    ]},
    { id: 2, name: 'In Progress', color: COL_COLORS[1], cards: [] },
    { id: 3, name: 'Done', color: COL_COLORS[2], cards: [] }
  ];
  state.nextColId = 4;
  state.nextCardId = 3;
}

// ── RENDER ──
function render() {
  const wrap = document.getElementById('board-wrap');
  const addBtn = document.getElementById('add-col-btn');
  wrap.innerHTML = '';

  state.columns.forEach(col => {
    wrap.appendChild(buildColumn(col));
  });

  wrap.appendChild(addBtn);

  document.getElementById('board-title-input').value = state.boardTitle;
  applyTheme();
  updateTaskCount();
}

function buildColumn(col) {
  const el = document.createElement('div');
  el.className = 'column';
  el.dataset.colId = col.id;

  const cardCount = col.cards.length;

  el.innerHTML = `
    <div class="col-header">
      <div class="col-dot" style="background:${col.color}"></div>
      <input class="col-name" value="${escHtml(col.name)}" maxlength="30"
        onchange="renameColumn(${col.id}, this.value)"
        onclick="event.stopPropagation()" />
      <span class="col-count">${cardCount}</span>
      <button class="col-menu-btn" onclick="deleteColumn(${col.id})" title="Delete column">×</button>
    </div>
    <div class="col-body" id="col-body-${col.id}"
      ondragover="onDragOver(event, ${col.id})"
      ondragleave="onDragLeave(event)"
      ondrop="onDrop(event, ${col.id})">
      ${col.cards.length === 0 ? '<div class="empty-col">Drop cards here</div>' : ''}
      ${col.cards.map(card => buildCardHTML(card, col.id)).join('')}
    </div>
    <div class="col-footer">
      <button class="add-card-btn" onclick="openAddCard(${col.id})">＋ Add card</button>
    </div>
  `;

  return el;
}

function buildCardHTML(card, colId) {
  const priorityColor = PRIORITY_COLORS[card.priority] || 'transparent';
  const dueStr = card.due ? formatDue(card.due) : '';
  const isOverdue = card.due && new Date(card.due) < new Date() && card.priority !== 'done';
  return `
    <div class="card" draggable="true" data-card-id="${card.id}" data-col-id="${colId}"
      ondragstart="onDragStart(event, ${card.id}, ${colId})"
      ondragend="onDragEnd(event)">
      <div class="card-priority-bar" style="background:${priorityColor}"></div>
      <div class="card-actions">
        <button class="card-action-btn" onclick="openEditCard(${card.id}, ${colId})" title="Edit">✎</button>
        <button class="card-action-btn delete" onclick="deleteCard(${card.id}, ${colId})" title="Delete">×</button>
      </div>
      <div class="card-title">${escHtml(card.title)}</div>
      ${card.desc ? `<div class="card-desc">${escHtml(card.desc)}</div>` : ''}
      <div class="card-footer">
        ${card.tag ? `<span class="card-tag">${escHtml(card.tag)}</span>` : ''}
        ${dueStr ? `<span class="card-due${isOverdue ? ' overdue' : ''}">${dueStr}</span>` : ''}
      </div>
    </div>
  `;
}

function formatDue(dateStr) {
  const d = new Date(dateStr);
  return d.toLocaleDateString('en-GB', { day: 'numeric', month: 'short' });
}

function escHtml(str) {
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function updateTaskCount() {
  const total = state.columns.reduce((sum, col) => sum + col.cards.length, 0);
  document.getElementById('task-count').textContent = `${total} task${total !== 1 ? 's' : ''}`;
}

// ── COLUMNS ──
function addColumn() {
  const limit = state.isPremium ? PREMIUM_COL_LIMIT : FREE_COL_LIMIT;
  if (state.columns.length >= limit) {
    showPremiumModal(`Free plan allows ${FREE_COL_LIMIT} columns. Upgrade for unlimited.`);
    return;
  }
  const col = {
    id: state.nextColId++,
    name: 'New Column',
    color: COL_COLORS[state.columns.length % COL_COLORS.length],
    cards: []
  };
  state.columns.push(col);
  save();
  render();
}

function deleteColumn(colId) {
  const col = state.columns.find(c => c.id === colId);
  if (!col) return;
  if (col.cards.length > 0 && !confirm(`Delete "${col.name}" and all ${col.cards.length} cards?`)) return;
  state.columns = state.columns.filter(c => c.id !== colId);
  save();
  render();
}

function renameColumn(colId, name) {
  const col = state.columns.find(c => c.id === colId);
  if (col) { col.name = name; save(); updateTaskCount(); }
}

// ── CARDS ──
let pendingColId = null;

function openAddCard(colId) {
  const col = state.columns.find(c => c.id === colId);
  const limit = state.isPremium ? PREMIUM_CARD_LIMIT : FREE_CARD_LIMIT;
  if (col && col.cards.length >= limit) {
    showPremiumModal(`Free plan allows ${FREE_CARD_LIMIT} cards per column. Upgrade for unlimited.`);
    return;
  }
  pendingColId = colId;
  editingCard = null;
  document.getElementById('card-modal-title').textContent = 'Add Card';
  document.getElementById('card-title-input').value = '';
  document.getElementById('card-desc-input').value = '';
  document.getElementById('card-priority-input').value = 'none';
  document.getElementById('card-due-input').value = '';
  document.getElementById('card-tag-input').value = '';
  document.getElementById('card-modal').classList.add('show');
  setTimeout(() => document.getElementById('card-title-input').focus(), 100);
}

function openEditCard(cardId, colId) {
  const col = state.columns.find(c => c.id === colId);
  if (!col) return;
  const card = col.cards.find(c => c.id === cardId);
  if (!card) return;
  pendingColId = colId;
  editingCard = cardId;
  document.getElementById('card-modal-title').textContent = 'Edit Card';
  document.getElementById('card-title-input').value = card.title;
  document.getElementById('card-desc-input').value = card.desc || '';
  document.getElementById('card-priority-input').value = card.priority || 'none';
  document.getElementById('card-due-input').value = card.due || '';
  document.getElementById('card-tag-input').value = card.tag || '';
  document.getElementById('card-modal').classList.add('show');
  setTimeout(() => document.getElementById('card-title-input').focus(), 100);
}

function closeCardModal() {
  document.getElementById('card-modal').classList.remove('show');
  editingCard = null;
  pendingColId = null;
}

function saveCard() {
  const title = document.getElementById('card-title-input').value.trim();
  if (!title) { document.getElementById('card-title-input').focus(); return; }

  const col = state.columns.find(c => c.id === pendingColId);
  if (!col) return;

  const cardData = {
    title,
    desc: document.getElementById('card-desc-input').value.trim(),
    priority: document.getElementById('card-priority-input').value,
    due: document.getElementById('card-due-input').value,
    tag: document.getElementById('card-tag-input').value.trim()
  };

  if (editingCard !== null) {
    const card = col.cards.find(c => c.id === editingCard);
    if (card) Object.assign(card, cardData);
  } else {
    col.cards.push({ id: state.nextCardId++, ...cardData });
  }

  save();
  render();
  closeCardModal();
}

function deleteCard(cardId, colId) {
  const col = state.columns.find(c => c.id === colId);
  if (!col) return;
  col.cards = col.cards.filter(c => c.id !== cardId);
  save();
  render();
}

// ── DRAG & DROP ──
function onDragStart(e, cardId, colId) {
  dragCard = cardId;
  dragCol = colId;
  setTimeout(() => {
    const el = document.querySelector(`[data-card-id="${cardId}"]`);
    if (el) el.classList.add('dragging');
  }, 0);
  e.dataTransfer.effectAllowed = 'move';
}

function onDragEnd(e) {
  document.querySelectorAll('.card').forEach(el => el.classList.remove('dragging'));
  document.querySelectorAll('.col-body').forEach(el => el.classList.remove('drag-over'));
}

function onDragOver(e, colId) {
  e.preventDefault();
  e.dataTransfer.dropEffect = 'move';
  document.querySelectorAll('.col-body').forEach(el => el.classList.remove('drag-over'));
  document.getElementById(`col-body-${colId}`).classList.add('drag-over');
}

function onDragLeave(e) {
  if (!e.currentTarget.contains(e.relatedTarget)) {
    e.currentTarget.classList.remove('drag-over');
  }
}

function onDrop(e, targetColId) {
  e.preventDefault();
  document.querySelectorAll('.col-body').forEach(el => el.classList.remove('drag-over'));

  if (dragCard === null) return;

  const srcCol = state.columns.find(c => c.id === dragCol);
  const tgtCol = state.columns.find(c => c.id === targetColId);
  if (!srcCol || !tgtCol) return;

  const cardIdx = srcCol.cards.findIndex(c => c.id === dragCard);
  if (cardIdx === -1) return;

  const [card] = srcCol.cards.splice(cardIdx, 1);

  // Insert at correct position
  const colBody = document.getElementById(`col-body-${targetColId}`);
  const cards = [...colBody.querySelectorAll('.card:not(.dragging)')];
  let insertIdx = tgtCol.cards.length;

  for (let i = 0; i < cards.length; i++) {
    const rect = cards[i].getBoundingClientRect();
    if (e.clientY < rect.top + rect.height / 2) {
      insertIdx = i;
      break;
    }
  }

  tgtCol.cards.splice(insertIdx, 0, card);
  dragCard = null;
  dragCol = null;
  save();
  render();
}

// ── SEARCH / FILTER ──
function filterCards(query) {
  const q = query.toLowerCase().trim();
  document.querySelectorAll('.card').forEach(card => {
    if (!q) { card.style.display = ''; return; }
    const title = card.querySelector('.card-title')?.textContent.toLowerCase() || '';
    const desc = card.querySelector('.card-desc')?.textContent.toLowerCase() || '';
    const tag = card.querySelector('.card-tag')?.textContent.toLowerCase() || '';
    card.style.display = (title.includes(q) || desc.includes(q) || tag.includes(q)) ? '' : 'none';
  });
}
